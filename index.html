function drawShareImage(title, desc, adv) { const canvas = document.getElementById('resCanvas'); const ctx = canvas.getContext('2d'); const s = 2; // 放大倍数提升清晰度 canvas.width = 400 * s; canvas.height = 650 * s; // 保持原有高度 ctx.scale(s, s); // ========================================== // 第一步：绘制复古纸张底色和边框 (最底层) // ========================================== ctx.fillStyle = "#f4f1ea"; ctx.fillRect(0, 0, 400, 650); ctx.strokeStyle = "#8b7e66"; ctx.lineWidth = 1; ctx.strokeRect(10, 10, 380, 630); // ========================================== // 第二步：绘制原本的诊断报告内容 (中间层) // ========================================== // 报告头 ctx.fillStyle = "#333"; ctx.font = "bold 22px Arial"; ctx.textAlign = "center"; ctx.fillText("2026 职场精神状态扫描报告", 200, 60); ctx.font = "bold 12px SimSun"; ctx.fillText("NO. " + Math.random().toString().slice(2, 10), 200, 85); // 诊断结果 (头衔) ctx.fillStyle = "#d32f2f"; ctx.font = "bold 32px sans-serif"; ctx.fillText(title, 200, 160); // 临床表现 ctx.fillStyle = "#555"; ctx.font = "bold 16px sans-serif"; ctx.fillText("【临床表现】", 200, 210); ctx.font = "14px sans-serif"; ctx.fillText(desc, 200, 240); // 医生建议 ctx.textAlign = "left"; ctx.fillStyle = "#333"; ctx.font = "bold 14px sans-serif"; ctx.fillText("【医生建议】", 40, 320); ctx.font = "13px sans-serif"; ctx.fillText(adv[0], 40, 350); ctx.fillText(adv[1], 40, 375); ctx.fillText(adv[2], 40, 400); // 原有印章 (y挪到了480，避免擋住裂變文案) ctx.beginPath(); ctx.strokeStyle = "rgba(211,47,47,0.7)"; ctx.lineWidth = 3; ctx.arc(320, 480, 45, 0, Math.PI * 2); // y坐标从500改为了480 ctx.stroke(); ctx.fillStyle = "rgba(211,47,47,0.7)"; ctx.font = "bold 16px Arial"; ctx.textAlign = "center"; ctx.fillText("状态合格", 320, 480); // y坐标同步修改 // ========================================== // 第三步：绘制底部的裂变文案 (最上层) // ========================================== // 绘制一个淡淡的分割线，让底部看起来更像一个功能区 ctx.beginPath(); ctx.strokeStyle = "rgba(139, 126, 102, 0.3)"; ctx.moveTo(30, 540); // 分割线在540 ctx.lineTo(370, 540); ctx.stroke(); // 挑衅裂变文案 (降维打击) ctx.textAlign = "center"; ctx.fillStyle = "#d32f2f"; // 搞成红色，加重裂变诱惑 ctx.font = "bold 18px sans-serif"; // 加粗红色大字 // 这里的 emoji "😏" 在 Canvas 中能否正常显示取决于用户的设备字体，通常手机端没问题 ctx.fillText("看看你的朋友，是什么货色 😏", 200, 565); // 移到分割线下方 // 回复引导文字 (y坐标调整) ctx.textAlign = "center"; ctx.fillStyle = "#666"; // 次要文字用灰色 ctx.font = "13px sans-serif"; ctx.fillText("关注公号【灵犀测评局】", 200, 580); // 扫码引导文字 (修改为小字、灰色) ctx.fillStyle = "#999"; ctx.font = "12px sans-serif"; // 搞小字，不要加粗 ctx.fillText("扫码回复 [不干了] 开启诊断报告", 200, 600); // 最后的版权声明 (y坐标调整) ctx.fillStyle = "#999"; ctx.font = "10px Arial"; ctx.fillText("©灵犀测评局", 200, 625); // ========================================== // 第四步：输出图片 // ========================================== setTimeout(() => { document.getElementById('share-img').src = canvas.toDataURL("image/png"); // 如果需要，可以在这里把loading文字隐藏掉 // document.getElementById('loading-text').style.display = 'none'; }, 300); }
